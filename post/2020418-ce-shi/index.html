	int x,y;

	{ return (x==a.x)?y<a.y:x<a.x; }

	{ return (P){x+a.x,y+a.y}; }

set<P> s;

int solve(P pos){
	if(s.count(pos)) return 0;
	s.insert(pos);
	int j=n,cnt=0;
	For(i,1,j){
		while(i<j && pos<a[i]+a[j]) ++cnt,--j;
		if(a[i]+a[j]==pos) --j;
		else ++cnt;
	}
	return cnt<=k;
}

	int ans=0;
	rd(n,k);

	For(i,1,k+1) Rof(j,n,n-k){
		ans+=solve(a[i]+a[j]);
	}
