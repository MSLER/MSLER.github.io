#include <bits/stdc++.h>
#define N 100005
#define ll long long
#define rep(i,x,y) for(i=(x);i<=(y);++i)
#define des(i,x,y) for(i=(x);i>=(y);--i)
#define ls(x,y) t[x].ls,t[y].ls,l,mid
#define rs(x,y) t[x].rs,t[y].rs,mid+1,r
using namespace std;

int a[N],pre[N],n,s[N][30][2],p[N],h[N],_1[N],cnt,rt[N];
struct node{
	int ls,rs,tag;
	ll sum;
} t[N<<7];

bool check(int l,int r){
	int i;
	des(i,29,0){
		int x=(pre[l-1]>>i)&1;
		if(s[r][i][x]>s[l-1][i][x]) return 0;
	} return 1;
}
void init(){
	int i,j;
	rep(i,1,n){
		pre[i]=pre[i-1]^a[i];
		des(j,29,0) if((a[i]>>j)&1){ _1[i]=j;break; }
	}
	des(i,29,0){
		int bit=0;
		rep(j,1,n){
			if((a[j]>>i)&1) bit^=1;
			s[j][i][0]=s[j-1][i][0];
			s[j][i][1]=s[j-1][i][1];
			if(i==_1[j] && (a[j]>>i)&1) s[j][i][bit]++;
		}
	}
	rep(i,1,n){
		int l=i,r=n;h[i]=i;
		while(l<=r){
			int mid=l+r>>1;
			if(check(i,mid)) h[i]=mid,l=mid+1;
			else r=mid-1;
		}
	}
}

void update(int &o,int d,int l,int r,int L,int R){
	o=++cnt,t[o]=t[d];
	t[o].sum+=R-L+1;
	if(L==l && r==R){ t[o].tag++;return; }	
	int mid=(l+r)>>1;
	if(R<=mid) update(ls(o,d),L,R);
	else if(L>mid) update(rs(o,d),L,R);
	else update(ls(o,d),L,mid),update(rs(o,d),mid+1,R);
}

ll query(int o,int d,int l,int r,int L,int R,int mrko,int mrkd){
	if(L==l && r==R) return t[o].sum-t[d].sum+(1ll*mrko-mrkd)*(1ll*R-L+1);
	int mid=l+r>>1;
	if(R<=mid) return query(ls(o,d),L,R,mrko+t[o].tag,mrkd+t[d].tag);
	else if(L>mid) return query(rs(o,d),L,R,mrko+t[o].tag,mrkd+t[d].tag);
	else return query(ls(o,d),L,mid,mrko+t[o].tag,mrkd+t[d].tag)
	+query(rs(o,d),mid+1,R,mrko+t[o].tag,mrkd+t[d].tag);
}

int main(){
	int i,q;
	ll lastans=0,l,r;
	scanf("%d",&n);
	rep(i,1,n) scanf("%d",&a[i]);
	init();
//	for(i=1;i<=n;++i) printf("%d : %d\n",i,h[i]);
	for(i=1;i<=n;++i) update(rt[i],rt[i-1],1,n,i,h[i]);
	scanf("%d",&q);
	while(q--){
		scanf("%lld%lld",&l,&r);
		l=(1ll*l+lastans)%n+1,r=(1ll*r+lastans)%n+1;
		if(l>r) swap(l,r);
		lastans=query(rt[r],rt[l-1],1,n,(int)l,(int)r,0,0);
		printf("%lld\n",lastans);
	}
}
